<script>

const firebaseConfig = {

  apiKey: "AIzaSyCZNvaM61VH4nEHIT21LWQtZrReUWg6WuE",

  authDomain: "jeopardy-tracker-b4e23.firebaseapp.com",

  databaseURL: "https://jeopardy-tracker-b4e23-default-rtdb.firebaseio.com",

  projectId: "jeopardy-tracker-b4e23"

};


firebase.initializeApp(firebaseConfig);

const database = firebase.database();

const scoresRef = database.ref("scores");


const baseValues = [200,400,600,800,1000];

let score = 0;

let multiplier = 1;


const board = document.getElementById("board");

const scoreDisplay = document.getElementById("score");


function buildBoard() {

  board.querySelectorAll("tr:not(:first-child)").forEach(r => r.remove());


  baseValues.forEach(v => {

    const row = document.createElement("tr");

    for (let i = 0; i < 6; i++) {

      const cell = document.createElement("td");

      cell.dataset.base = v;

      cell.dataset.state = 0;

      cell.textContent = `$${v * multiplier}`;

      cell.onclick = () => handleClick(cell);

      row.appendChild(cell);

    }

    board.appendChild(row);

  });

}


function handleClick(cell) {

  const value = Number(cell.dataset.base) * multiplier;

  const state = Number(cell.dataset.state);


  if (state === 0) {

    score += value;

    cell.className = "correct";

    cell.dataset.state = 1;

  } else if (state === 1) {

    score -= value * 2;

    cell.className = "wrong";

    cell.dataset.state = 2;

  } else {

    score += value;

    cell.className = "";

    cell.dataset.state = 0;

  }

  scoreDisplay.textContent = `Score: $${score}`;

}


document.getElementById("roundToggle").onclick = () => {

  multiplier = multiplier === 1 ? 2 : 1;

  buildBoard();

};


document.getElementById("resetScore").onclick = () => {

  score = 0;

  scoreDisplay.textContent = "Score: $0";

  buildBoard();

};


document.getElementById("submitScore").onclick = () => {

  const name = playerName.value.trim();

  const episodeDate = episodeDate.value.trim();

  if (!name || !episodeDate) return alert("Missing info");


  scoresRef.push({

    name,

    score,

    episodeDate,

    submittedDate: new Date().toLocaleString(),

    timestamp: Date.now()

  });


  score = 0;

  scoreDisplay.textContent = "Score: $0";

  buildBoard();

};


function showPage(e, page) {

  document.querySelectorAll(".nav-btn").forEach(b => b.classList.remove("active"));

  e.target.classList.add("active");


  document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));

  document.getElementById(page === "game" ? "gamePage" : "leaderboardPage")

    .classList.add("active");


  if (page === "leaderboard") loadLeaderboard();

}


function loadLeaderboard() {

  const list = document.getElementById("scoresList");

  scoresRef.once("value", snap => {

    const data = snap.val();

    if (!data) {

      list.innerHTML = '<p class="no-scores">No scores yet.</p>';

      return;

    }

    list.innerHTML = Object.values(data)

      .sort((a,b) => b.timestamp - a.timestamp)

      .map(s => `

        <div class="score-entry">

          <div>${s.name}<br>${s.episodeDate}</div>

          <div>$${s.score}</div>

        </div>

      `).join("");

  });

}


buildBoard();

</script>
